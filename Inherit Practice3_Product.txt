package com.kh.hw3.model.vo;

import javax.print.attribute.standard.MediaSize.Other;

public class Product {
	private String brand;
	private String name;
	private String color;
	private int price;
	
	public Product() {}

	public Product(String brand, String name, String color, int price) {
		super();
		this.brand = brand;
		this.name = name;
		this.color = color;
		this.price = price;
	}

	public String getBrand() {
		return brand;
	}

	public void setBrand(String brand) {
		this.brand = brand;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public String getColor() {
		return color;
	}

	public void setColor(String color) {
		this.color = color;
	}

	public int getPrice() {
		return price;
	}

	public void setPrice(int price) {
		this.price = price;
	}
	
	@Override
	public String toString() {
		return "brand=" + brand + ", name=" + name + ", color=" + color + ", price=" + price;
		
	}
	
	@Override
	public boolean equals(Object obj) {
		Product other = (Product)obj;
		if(brand.equals(other.getBrand()) && name.equals(other.getName()) && color.equals(other.getColor()) && price == other.getPrice()) {
			return true;
		}
		return false;
	}
	
	@Override
	public int hashCode() {
		return (brand + name + color + price).hashCode();
		
	}
	
	@Override
	public Product clone() {
		return new Product(brand, name, color, price);
	}
	
	
	
	
	
	
	
}
